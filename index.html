1. Introdução

Este relatório descreve o projeto de controle de um servo motor utilizando um sensor ultrassônico e um Arduino, com o objetivo de desenvolver habilidades em eletrônica e programação.

2. Objetivos

O principal objetivo é fazer o controle da direção do servo motor com a utilização de um sensor ultrassônico em um microcontrolador Arduino.

3. Materiais Utilizados

  • Arduino
  • 1 Sensor ultrassônico
  • 1 Servo motor
  • Protoboard
  • Fios de conexão

5. Código Fonte


#include 

int cm = 0;
long distancia, duracao;
int posicao1 = 180;
int posicao2 = 1;
Servo meuServo;

const int Pecho = 7; 
const int Ptrig = 8; 

void setup() {
  Serial.begin(9600);
  pinMode(Pecho, INPUT);
  pinMode(Ptrig, OUTPUT);
  meuServo.attach(13);
}

void loop() {
  digitalWrite(Ptrig, LOW);
  delayMicroseconds(2);
  digitalWrite(Ptrig, HIGH);
  delayMicroseconds(10); 
  digitalWrite(Ptrig, LOW);

  duracao = pulseIn(Pecho, HIGH);
  distancia = (duracao / 2) / 29.1;
  
  float ang = map(distancia, 50, 230, 0, 180);
  Serial.println(ang);

  meuServo.write(constrain(ang, 0, 180)); 
  delay(400);
}
